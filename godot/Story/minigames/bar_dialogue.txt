# Мини-игра: Диалоги в баре с Димой

mark minigame_bar_dialogue

background polish_home fade_in

"=== МИНИ-ИГРА: ДИАЛОГИ В БАРЕ ==="

"Вы входите в бар 'У Михалыча'."

dima happy enter right "О! Данила! Давно не виделись!"

danila happy "Привет, Дима! Как дела?"

dima neutral "Да нормально. Работа, дом... Ты как?"

# Раунд 1: Установление контакта
choice:
	"Предложить выпить":
		danila happy "Давай пропустим по пиву?"
		dima happy "Вот это дело! Михалыч, два пива!"
		
		set friendship_dima 20
		"Дружба с Димой +5"
		
	"Сразу спросить о заводе":
		danila neutral "Слушай, а как там на заводе работается?"
		dima suspicious "Что-то ты сразу про завод... Странно."
		
		set stealth_level 48
		set suspicion_volkov 28
		"Скрытность -2 (слишком явно)"
		"Подозрения +8"
		jump bar_dialogue_failed
	
	"Поговорить о жизни":
		danila neutral "Как жизнь вообще? Нравится работа?"
		dima neutral "Работа как работа. Скучновато, но платят нормально."
		
		set friendship_dima 18
		"Дружба +3"

# Выпивка
"Вы пьете пиво с Димой."

dima neutral "Слушай, а в Польше как? Лучше чем у нас?"

# Раунд 2: Разговор о странах
choice:
	"Сказать что везде одинаково":
		danila neutral "Да везде примерно одинаково. Люди везде люди."
		dima happy "Правильно говоришь!"
		
		set friendship_dima 23
		set cultural_knowledge 12
		"Дружба +5"
		
	"Сказать что в Польше лучше":
		danila neutral "Ну, зарплаты повыше, жизнь подороже."
		dima sad "Везет вам..."
		
		set friendship_dima 19
		"Дружба +1 (не очень дипломатично)"
		
	"Похвалить Россию":
		danila happy "У вас своя атмосфера. Настоящая душа."
		dima happy "Точно! У нас душа есть!"
		
		set friendship_dima 25
		set stealth_level 53
		"Дружба +7"
		"Скрытность +3 (хорошая дипломатия)"

# После нескольких кружек
"Несколько кружек пива спустя..."
"Дима явно подвыпил."

dima drunk "Слушай, Данила... Ты хороший мужик!"

danila happy "Ты тоже, Дима!"

dima thinking "А знаешь... На заводе странные дела..."

danila thinking "(Вот оно! Пора действовать осторожно)"

# Раунд 3: Получение информации
choice:
	"Осторожно спросить что за дела":
		danila neutral "Что за дела?"
		
		dima drunk "Ну... Такое... Секретное все..."
		dima thinking "Директор новый приехал. Все засекретил."
		
		"Вы получили полезную информацию!"
		
		set friendship_dima 28
		set stealth_level 56
		set mission2_quality 85
		
		"Дружба +10"
		"Скрытность +6 (осторожный подход)"
		"Задание на 85%"
		
	"Спросить прямо про завод":
		danila serious "Дима, что там на заводе происходит?"
		
		dima suspicious "Погоди... А зачем тебе?"
		dima thinking "Ты же бизнесмен... Почему так интересуешься?"
		
		choice:
			"Инвестиции же" if stealth_level > 50:
				danila neutral "Ну я же хочу инвестировать. Нужно знать."
				dima neutral "А, ну да... Логично."
				
				set friendship_dima 24
				set mission2_quality 70
				"Дружба +6"
				"Задание на 70%"
				
			"Просто любопытно":
				danila neutral "Просто интересно."
				dima suspicious "Хм... Странно ты какой-то..."
				
				set friendship_dima 20
				set stealth_level 47
				set suspicion_volkov 32
				set mission2_quality 55
				
				"Скрытность -3"
				"Подозрения +12"
				"Задание на 55%"
	
	"Не спрашивать, просто пить":
		danila neutral "Расскажешь, когда будешь готов."
		
		dima happy "Ты понимаешь! Не как эти... что все про работу спрашивают."
		
		"Вы завоевали доверие, но информации получили мало."
		
		set friendship_dima 30
		set stealth_level 54
		set mission2_quality 60
		
		"Дружба +12 (отличные отношения)"
		"Скрытность +4"
		"Задание на 60%"

# Конец вечера
mark bar_dialogue_end

"Вечер подходит к концу."

dima drunk "Данила... Ты мой друг!"

danila happy "И ты мой, Дима!"

dima happy "Давай еще встретимся!"

danila neutral "Обязательно!"

"Вы покидаете бар."

danila thinking "(Контакт установлен. Но нужно быть осторожным.)"
